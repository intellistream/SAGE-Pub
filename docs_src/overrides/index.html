# 安装 SAGE
pip install sage-framework

# 创建简单的RAG管道
from sage import Environment, FileSource, DenseRetriever, QAPromptor

pipeline = Environment(name="my_rag_pipeline")
result = (pipeline
    .from_source(FileSource, config)
    .map(DenseRetriever, config)
    .map(QAPromptor, config)
    .execute()
)
